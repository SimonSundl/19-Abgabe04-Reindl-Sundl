3   public interface Queue {
4   	
5   	/**
6   	 *  Adds the element obj to the queue. 
7   	 *  If the addition is successful, the method returns true else false.
8   	 * @param obj
9   	 * @return
10  	 */
11  	public abstract boolean offer(String obj);
12  
13  	/**
14  	 *  Returns the head (first) element and also deletes it. That is, we cannot get it again. 
15  	 *  If no element exists (when queue is empty), the method returns null.
16  	 * @return
17  	 */
18  	public abstract String poll();
19  	
20  	/**
21  	 * It also returns and deletes the head element like poll(), but with a small difference. 
22  	 * This method throws NoSuchElementException if the queue is empty. 
23  	 * 
24  	 * @return
25  	 */
26  	public abstract String remove();
27  	
28  	/**
29  	 *  Returns the head element but it does not delete it. That is, we can get it again. 
30  	 *  Returns null when the queue is empty. 
31  	 * @return
32  	 */
33  	public abstract String peek();
34  	
35  	/**
36  	 *  It works similar to peek() but with a small difference (returns but does not delete the element). 
37  	 *  It throws NoSuchElementException when the queue is empty.
38  	 * @return
39  	 */
40  	public abstract String element();
41  	
42  }
