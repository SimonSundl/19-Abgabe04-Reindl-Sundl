24      @Test
25      public void simple(){
26          test.offer("Hi");
27          test.offer("Hello");
28          test.offer("Ciao");
29          test.offer("Hallo");
30          test.offer("Servus");
31          Assert.assertEquals("Hi", test.remove());
32          Assert.assertEquals("Hello", test.remove());

34          Assert.assertEquals("Hallo", test.remove());
35          Assert.assertEquals("Servus", test.remove());
36      }
37  
38      /**
39       * Forcing NoSuchElementException whilst trying to remove an object from an empty queue.

41      @Test(expected = NoSuchElementException.class)

44      }
45  
46      /**
47       * Testing if maximum size works as intended.
48       */
49      @Test
50      public void size(){
51          StringQueue test2 = new StringQueue(1);
52          test2.offer("Testing");
53          Assert.assertEquals(1, test2.getMaxSize());
54          Assert.assertEquals(5, test.getMaxSize());
55          Assert.assertEquals("Testing", test2.poll());
56      }
